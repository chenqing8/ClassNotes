#### 构造器模式

```js
// es5
function Student (name, age) {
  this.name = name;
  this.age = age;
  this.eating = function () {
    console.log(this.name + '去吃饭啦！！！');
  }
}
var cq = new Student('cq', 25);
var ng = new Student('ng', 21);
ng.eating();



// es6
class Student {
  constructor(name, age) {
    this.name = name;
    this.age = age;
  }
  eating () {
    console.log(this.name + '去吃饭啦！！！');
  }
}
var cq = new Student('cq', 25);
var ng = new Student('ng', 21);
cq.eating();
```
